param(
    [string]$EnvFile = ".env",
    [int]$Port = 8080
)

Set-StrictMode -Version Latest
$ErrorActionPreference = "Stop"

if (-not (Test-Path ".venv")) {
    py -3.12 -m venv .venv
}

.\.venv\Scripts\python.exe -m pip install --upgrade pip
.\.venv\Scripts\pip.exe install -r requirements.txt

. .\scripts\set-env.ps1 -Path $EnvFile
$env:PORT = "$Port"

.\.venv\Scripts\python.exe -m uvicorn data_a2a_agent.agent:a2a_app --host 0.0.0.0 --port $Port
