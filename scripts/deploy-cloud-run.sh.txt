#!/usr/bin/env bash
set -euo pipefail

: "${GOOGLE_CLOUD_PROJECT:?Set GOOGLE_CLOUD_PROJECT}"
: "${GOOGLE_CLOUD_LOCATION:?Set GOOGLE_CLOUD_LOCATION}"
: "${BIGQUERY_TABLES:?Set BIGQUERY_TABLES}"
: "${GCS_BUCKET_NAME:?Set GCS_BUCKET_NAME}"

SERVICE_NAME="${SERVICE_NAME:-enterprise-data-a2a-agent}"
AGENT_MODEL="${AGENT_MODEL:-gemini-2.0-flash}"
GOOGLE_GENAI_USE_VERTEXAI="${GOOGLE_GENAI_USE_VERTEXAI:-true}"
BIGQUERY_TABLE_ALIASES="${BIGQUERY_TABLE_ALIASES:-}"
BIGQUERY_MAX_BYTES_BILLED="${BIGQUERY_MAX_BYTES_BILLED:-1000000000}"
BIGQUERY_DEFAULT_LIMIT="${BIGQUERY_DEFAULT_LIMIT:-100}"
BIGQUERY_MAX_LIMIT="${BIGQUERY_MAX_LIMIT:-1000}"
GCS_PREFIX="${GCS_PREFIX:-}"
GCS_MAX_TEXT_BYTES="${GCS_MAX_TEXT_BYTES:-200000}"
GCS_MAX_PDF_BYTES="${GCS_MAX_PDF_BYTES:-25000000}"
GCS_MAX_PDF_PAGES="${GCS_MAX_PDF_PAGES:-20}"
GCS_MAX_PDF_TEXT_CHARS="${GCS_MAX_PDF_TEXT_CHARS:-60000}"
A2A_PUBLIC_URL="${A2A_PUBLIC_URL:-}"
ALLOW_UNAUTHENTICATED="${ALLOW_UNAUTHENTICATED:-false}"

ENV_FILE="$(mktemp)"
cleanup() {
  rm -f "$ENV_FILE"
}
trap cleanup EXIT

append_env() {
  local key="$1"
  local value="$2"
  value="${value//\\/\\\\}"
  value="${value//\"/\\\"}"
  printf '%s: "%s"\n' "$key" "$value" >> "$ENV_FILE"
}

append_env "GOOGLE_CLOUD_PROJECT" "$GOOGLE_CLOUD_PROJECT"
append_env "GOOGLE_CLOUD_LOCATION" "$GOOGLE_CLOUD_LOCATION"
append_env "GOOGLE_GENAI_USE_VERTEXAI" "$GOOGLE_GENAI_USE_VERTEXAI"
append_env "AGENT_MODEL" "$AGENT_MODEL"
append_env "BIGQUERY_TABLES" "$BIGQUERY_TABLES"
append_env "BIGQUERY_TABLE_ALIASES" "$BIGQUERY_TABLE_ALIASES"
append_env "BIGQUERY_MAX_BYTES_BILLED" "$BIGQUERY_MAX_BYTES_BILLED"
append_env "BIGQUERY_DEFAULT_LIMIT" "$BIGQUERY_DEFAULT_LIMIT"
append_env "BIGQUERY_MAX_LIMIT" "$BIGQUERY_MAX_LIMIT"
append_env "GCS_BUCKET_NAME" "$GCS_BUCKET_NAME"
append_env "GCS_PREFIX" "$GCS_PREFIX"
append_env "GCS_MAX_TEXT_BYTES" "$GCS_MAX_TEXT_BYTES"
append_env "GCS_MAX_PDF_BYTES" "$GCS_MAX_PDF_BYTES"
append_env "GCS_MAX_PDF_PAGES" "$GCS_MAX_PDF_PAGES"
append_env "GCS_MAX_PDF_TEXT_CHARS" "$GCS_MAX_PDF_TEXT_CHARS"
append_env "A2A_PUBLIC_URL" "$A2A_PUBLIC_URL"

AUTH_FLAG="--no-allow-unauthenticated"
if [[ "$ALLOW_UNAUTHENTICATED" == "true" ]]; then
  AUTH_FLAG="--allow-unauthenticated"
fi

gcloud run deploy "$SERVICE_NAME" \
  --source . \
  --region "$GOOGLE_CLOUD_LOCATION" \
  --project "$GOOGLE_CLOUD_PROJECT" \
  "$AUTH_FLAG" \
  --env-vars-file "$ENV_FILE"
